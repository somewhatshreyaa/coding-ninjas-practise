#include<iostream>
using namespace std;

bool prime(int num){
	if(num<=1){
		return false;
	}
	
	for(int i=2; i<(num); i++){
		if(num%i==0){
            return false;
		}
		
		
	}
	return true;




}

int main() {
	// Write your code here
	int num;
	cin>>num;
	// bool ans;
    if (prime(num)==true){
		cout<<"true";
	}
	else{
		cout<<"false";
	}





	
	
	
}
