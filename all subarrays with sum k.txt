
#include <vector>
using namespace std;

int findAllSubarraysWithGivenSum(vector<int>& arr, int k) {
    int n = arr.size();
    int left = 0, right = 0;
    int sum = arr[0];
    int count = 0;

    while (left < n && right < n) {
        if (sum == k) {
            count++;
            sum -= arr[left];
            left++;
        } else if (sum < k) {
            right++;
            if (right < n) {
                sum += arr[right];
            }
        } else {
            sum -= arr[left];
            left++;
        }
    }

    return count;
}
