public class Solution {

    public static int NthRoot(int n, int m) {

        // Write your code here.

        double result = Math.pow(m,1.0/n);

        int ans = (int) Math.round(result);

        if(Math.pow(ans,n)==m){

            return ans;
        }
        else{

            return -1;

        }

    }
}
