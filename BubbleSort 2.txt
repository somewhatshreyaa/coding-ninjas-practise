void bubbleSort(int arr[], int n){
    int temp = 0;
    
    for( int i = 0 ; i < n-1; i++){
        int j = i;
            while (j >= 0 && arr[j] > arr[j + 1]){
                temp = arr[j];

                arr[j] = arr[j+1];

                arr[j+1] = temp;

                j--;


            }

    }
}
